git status    - 查看仓库状态
git add       - 添加文件到暂存区
git commit    - 提交更改
git log       - 查看提交历史
git branch    - 查看分支
git checkout  - 切换分支
git pull      - 从远程仓库拉取更新
git push      - 推送到远程仓库

git checkout -b <新分支名称> 创建并切换到新分支

git merge <分支名称> 合并分支
